#!/bin/bash

# ==========================================
# 💜 Reaction Generator Script
# ------------------------------------------
# Input  : $type (commit type)  ← via ENV
#          $mode (time mode)    ← via ENV
#
# Output : reaction text (stdout)
#
# Flow:
# validate → select reaction → fallback → output
# ==========================================


# =========================
# 🔹 VALIDATION
# ✅ FIX: validasi $type dan $mode sebelum dipakai
# BUG LAMA: langsung dipakai tanpa cek → silent wrong behavior
# =========================
if [ -z "$type" ]; then
  echo "[ERROR] \$type not set" >&2
  exit 1
fi

if [ -z "$mode" ]; then
  echo "[ERROR] \$mode not set" >&2
  exit 1
fi


# =========================
# 📦 LOAD DATA
# =========================
REACTION_FILE=".github/messages/reaction.json"

if [ ! -f "$REACTION_FILE" ]; then
  echo "[ERROR] reaction.json not found: $REACTION_FILE" >&2
  exit 1
fi

echo "[LOG] Loading reaction.json" >&2


# =========================
# 🧠 INPUT INFO
# =========================
echo "[LOG] Input → type=$type | mode=$mode" >&2


# =========================
# 🎯 SELECT REACTION
# ✅ FIX: jq baca langsung dari file, tidak lewat cat+variable+echo
# BUG LAMA: echo "$data" bisa mangle escape sequences di JSON
# =========================
reaction=$(jq -r \
  --arg type "$type" \
  --arg mode "$mode" \
  '.[$type][$mode][]?' "$REACTION_FILE" | shuf -n 1)

echo "[LOG] Raw reaction: $reaction" >&2


# =========================
# ⚠️ FALLBACK SYSTEM
# Fallback berlapis biar gak pernah kosong
# =========================
if [ -z "$reaction" ] || [ "$reaction" = "null" ]; then
  echo "[LOG] Primary reaction empty → fallback: update" >&2

  reaction=$(jq -r \
    --arg mode "$mode" \
    '.update[$mode][]?' "$REACTION_FILE" | shuf -n 1)
fi

if [ -z "$reaction" ] || [ "$reaction" = "null" ]; then
  echo "[LOG] Secondary fallback → default message" >&2
  reaction="kamu tetap jalan hari ini… itu udah cukup"
fi


# =========================
# ✅ FINAL VALIDATION
# =========================
if [ -z "$reaction" ]; then
  echo "[ERROR] Reaction still empty → exit" >&2
  exit 1
fi


# =========================
# 📤 OUTPUT
# =========================
echo "[LOG] Final reaction ready" >&2
echo "$reaction"

#!/bin/bash

# ==========================================
# 💜 Bot Engine Core
# ------------------------------------------
# Fungsi:
# Engine utama untuk generate & kirim pesan
# berdasarkan MODE + optional TAG
#
# Input (ENV):
# MODE  → ambitious / productive / warning dll
# TAG   → feat / fix / docs dll (optional)
#
# Output:
# Kirim pesan dari bot config
#
# Flow:
# 1. Load config JSON
# 2. Loop semua bot
# 3. Cari message sesuai MODE
# 4. Optional: inject TAG context
# 5. Random greeting + message
# 6. Delay per bot
# 7. Kirim ke Telegram
# ==========================================

CONFIG=".github/scripts/agent/agent.config.json"

send_message () {
  curl -s -X POST "https://api.telegram.org/bot$TELEGRAM_TOKEN/sendMessage" \
    -d chat_id="$TELEGRAM_CHANNEL_ID" \
    --data-urlencode "text=$1"
}

echo "[BOT] START ENGINE"

# =========================
# 🔹 VALIDATION
# =========================
if [ -z "$MODE" ]; then
  echo "[ERROR] MODE not set" >&2
  exit 1
fi

if [ ! -f "$CONFIG" ]; then
  echo "[ERROR] Config not found: $CONFIG" >&2
  exit 1
fi

echo "[BOT] MODE: $MODE"
echo "[BOT] TAG : ${TAG:-<none>}"

# ✅ FIX: 3 seed terpisah dan independen
# BUG LAMA: RANDOM_SEED=$RANDOM lalu seed*1000 di jq
# → 1000 habis dibagi 4 dan 8 (ukuran array di config)
# → (seed * 1000) % 4 = 0 SELALU, (seed * 1000) % 8 = 0 SELALU
# → bot SELALU pilih index 0 untuk tone, greeting, dan message
SEED_TONE=$RANDOM
SEED_GREET=$RANDOM
SEED_MSG=$RANDOM

echo "[BOT] SEEDS: tone=$SEED_TONE greet=$SEED_GREET msg=$SEED_MSG"

bots=$(jq -r 'keys[]' "$CONFIG")

for bot in $bots; do

  delay=$(jq -r --arg bot "$bot" '.[$bot].delay // 0' "$CONFIG")

  reply=$(jq -r \
    --arg bot "$bot" \
    --arg mode "$MODE" \
    --arg tag "${TAG:-}" \
    --argjson seed_tone "$SEED_TONE" \
    --argjson seed_greet "$SEED_GREET" \
    --argjson seed_msg "$SEED_MSG" \
    '
    .[$bot].message as $root

    # =========================
    # 🔹 FIND CATEGORY
    # =========================
    | ($root | to_entries
      | map(select(.value[$mode]))
      | .[0]) as $category

    | if $category == null then empty else

      ($category.value[$mode]) as $group

      # =========================
      # 🔹 RANDOM TONE
      # ✅ FIX: $seed_tone % length (tanpa *1000)
      # =========================
      | ($group | keys) as $toneKeys
      | $toneKeys[ $seed_tone % ($toneKeys | length) ] as $tone
      | ($group[$tone]) as $data

      # =========================
      # 🔹 BASE MESSAGE
      # ✅ FIX: seed terpisah untuk greet dan msg
      # =========================
      | $data.greetings[
          $seed_greet % ($data.greetings | length)
        ] as $greet

      | $data.messages[
          $seed_msg % ($data.messages | length)
        ] as $msg

      # =========================
      # 🔹 TAG INJECTION (OPTIONAL)
      # =========================
      | if ($tag != null and $tag != "") then
          ($root.reaction[$tag][$mode] // []) as $tagData
          | if ($tagData | length) > 0 then
              $msg + "\n\n" +
              $tagData[ $seed_msg % ($tagData | length) ]
            else
              $msg
            end
        else
          $msg
        end

      # =========================
      # 🔹 FINAL BUILD
      # =========================
      | $greet + "\n\n" + .
    end

    ' "$CONFIG")

  if [ -n "$reply" ]; then
    echo "[BOT] $bot → send in ${delay}s"
    (
      sleep "$delay"
      send_message "$reply"
    ) &
  else
    echo "[BOT] $bot → skip (no match for mode: $MODE)"
  fi

done

wait
echo "[BOT] DONE 💜"

#!/bin/bash

# ==========================================
# 💜 Message Generator Script
# ------------------------------------------
# Input  : $mode_message (json file name)
#          $mode (time / condition mode)
#
# Output :
# TYPE=combined → TEXT
# TYPE=split    → GREETING + MESSAGE
#
# Flow:
# load json → pick tone → pick greeting/message
#           → validate → random output format
# ==========================================


# =========================
# 📦 LOAD DATA
# =========================
echo "[LOG] Loading message file: $mode_message.json" >&2
data=$(cat .github/messages/"$mode_message".json)


# =========================
# 🎭 SELECT RANDOM TONE
# =========================
echo "[LOG] Selecting tone for mode: $mode" >&2
tone=$(echo "$data" | jq -r --arg mode "$mode" '
  .[$mode] | keys[]' | shuf -n 1)

echo "[LOG] Selected tone: $tone" >&2


# =========================
# 👋 SELECT GREETING
# =========================
greeting=$(echo "$data" | jq -r \
  --arg mode "$mode" \
  --arg tone "$tone" '
  .[$mode][$tone].greetings[]' | shuf -n 1)

echo "[LOG] Selected greeting: $greeting" >&2


# =========================
# 💬 SELECT MESSAGE
# =========================
message=$(echo "$data" | jq -r \
  --arg mode "$mode" \
  --arg tone "$tone" '
  .[$mode][$tone].messages[]' | shuf -n 1)

echo "[LOG] Selected message: $message" >&2


# =========================
# ✅ VALIDATION
# =========================
if [ -z "$tone" ] || [ -z "$greeting" ] || [ -z "$message" ]; then
  echo "[ERROR] No message found → exit" >&2
  exit 1
fi


# =========================
# 🎲 OUTPUT MODE (GACHA)
# =========================
gacha=$((RANDOM % 2))
echo "[LOG] Gacha result: $gacha" >&2

if [ "$gacha" -eq 0 ]; then
  echo "[LOG] Output mode: combined" >&2
  echo "TYPE=combined"
  echo "TEXT=$greeting $message"
else
  echo "[LOG] Output mode: split" >&2
  echo "TYPE=split"
  echo "GREETING=$greeting"
  echo "MESSAGE=$message"
fi

#!/bin/bash

# ==========================================
# 💜 Telegram Smart Sender (Safe Mode)
# ------------------------------------------
# Fungsi:
# Mengirim pesan ke Telegram dengan mode:
# - combined → 1 pesan
# - split    → greeting + message
# - auto     → random (gacha)
#
# Behavior:
# - Tidak akan fail workflow (safe exit)
# - Skip jika data tidak valid
# - Mendukung fallback combine
#
# ENV:
#   DELIVERY_MODE → combined / split / auto
#   TEXT
#   GREETING
#   MESSAGE
#   TELEGRAM_TOKEN
#   TELEGRAM_CHANNEL_ID
#
# Output:
#   Log terstruktur + status pengiriman
#
# ==========================================


# =========================
# 🔹 LOG SYSTEM
# =========================
log_info() {
  echo "[INFO] $1" >&2
}

log_warn() {
  echo "[WARN] $1" >&2
}

log_error() {
  echo "[ERROR] $1" >&2
}

log_debug() {
  echo "[DEBUG] $1" >&2
}


# =========================
# 🔹 SAFE EXIT (NO FAIL)
# =========================
safe_exit () {
  log_warn "$1"
  return 0 2>/dev/null || exit 0
}


# =========================
# 🔹 SEND FUNCTION
# =========================
send_message () {
  local text="$1"

  log_debug "Sending payload..."
  
  response=$(curl -s -w "%{http_code}" -o /dev/null -X POST \
    "https://api.telegram.org/bot$TELEGRAM_TOKEN/sendMessage" \
    -d chat_id="$TELEGRAM_CHANNEL_ID" \
    --data-urlencode "text=$text")

  if [ "$response" != "200" ]; then
    log_warn "Telegram API responded with HTTP $response"
  else
    log_info "Message sent successfully"
  fi
}


# =========================
# 🚀 START
# =========================
log_info "Smart delivery started"


# =========================
# 🔹 VALIDATION BASE
# =========================
if [ -z "$TELEGRAM_TOKEN" ] || [ -z "$TELEGRAM_CHANNEL_ID" ]; then
  safe_exit "Missing Telegram credentials → skip"
fi

[ -z "$DELIVERY_MODE" ] && DELIVERY_MODE="combined"

log_info "Delivery mode: $DELIVERY_MODE"


# =========================
# 🔹 AUTO MODE (GACHA)
# =========================
if [ "$DELIVERY_MODE" = "auto" ]; then
  rand=$((RANDOM % 100))

  if [ "$rand" -lt 50 ]; then
    MODE="combined"
  else
    MODE="split"
  fi

  log_info "Auto selected mode → $MODE (roll=$rand)"

else
  MODE="$DELIVERY_MODE"
  log_info "Fixed mode → $MODE"
fi


# =========================
# 📤 SEND LOGIC
# =========================
if [ "$MODE" = "combined" ]; then

  log_info "Executing combined mode"

  if [ -z "$TEXT" ]; then
    if [ -n "$GREETING" ] && [ -n "$MESSAGE" ]; then
      TEXT="$GREETING

$MESSAGE"
      log_debug "Fallback: combine greeting + message"
    else
      safe_exit "No valid content for combined → skip"
    fi
  fi

  send_message "$TEXT"


else

  log_info "Executing split mode"

  if [ -z "$GREETING" ] || [ -z "$MESSAGE" ]; then
    safe_exit "Missing GREETING or MESSAGE → skip"
  fi

  log_debug "Sending greeting"
  send_message "$GREETING"

  delay=$((RANDOM % 5 + 2))
  log_info "Delay before main message: ${delay}s"
  sleep $delay

  log_debug "Sending main message"
  send_message "$MESSAGE"

fi


# =========================
# ✅ DONE
# =========================
log_info "Smart delivery finished 💜"

# ==========================================

get_mode_by_commit() {

  local hour=$(date +%H)
  local mode=""

  if [ "$hour" -ge 12 ] && [ "$hour" -lt 16 ]; then
    mode="ambitious"
  elif [ "$hour" -ge 16 ] && [ "$hour" -lt 18 ]; then
    mode="productive"
  elif [ "$hour" -ge 18 ] && [ "$hour" -lt 21 ]; then
    mode="persistent"
  elif [ "$hour" -ge 21 ] || [ "$hour" -lt 7 ]; then
    mode="consistent"
  fi

  # ❗ kalau kosong → jangan echo sampah
  if [ -z "$mode" ]; then
    echo "No mode detected" >&2
    return 1
  fi

  echo "$mode"
}

  export TZ=Asia/Jakarta  
  local hour=$(date +%H)

  echo "⏳ Hour: $hour" >&2

  local mode=""
  local send=false
  
  if [ "$hour" -ge 12 ] && [ "$hour" -lt 19 ]; then
    [ $((RANDOM % 4)) -eq 0 ] && send=true
    mode="warning" 
    echo "🌤️ Phase: Sore (low frequency)" >&2
    
  elif [ "$hour" -ge 19 ] && [ "$hour" -lt 23 ]; then
    [ $((RANDOM % 3)) -eq 0 ] && send=true
    mode="last_warning"  
    echo "🌙 Phase: Malam (medium)" >&2

    [ $((RANDOM % 2)) -eq 0 ] && send=true
    mode="final_warning"  
    echo "🌌 Phase: Late (high)" >&2
  fi
  
  if [ "$hour" -lt 12 ]; then
    echo "⏭️ Skip (smart interval)"
    echo "skip=true" >> $GITHUB_ENV
    return 0
  fi

  echo "$mode"
}

#!/bin/bash

# ==========================================
# 💜 Commit Parser Core
# ------------------------------------------
# Input  :
#   COMMIT_MESSAGE
#
# Output :
#   type
#   detail
#   reaction
#
# Usage:
#   source parse.commit.sh
#   parse_commit "$msg"
# ==========================================

parse_commit() {
  local msg="$1"

  # =========================
  # 🔹 PARSE TYPE & DETAIL
  # =========================
  if [[ "$msg" == *":"* ]]; then
    TYPE=$(echo "$msg" | head -n1 | cut -d':' -f1)
    DETAIL=$(echo "$msg" | tail -n +1 | cut -d':' -f2-)
  else
    TYPE="update"
    DETAIL="$msg"
  fi

  # =========================
  # 🔹 MAP REACTION
  # =========================
  case "$TYPE" in
    feat*) REACTION="feat" ;;
    fix*) REACTION="fix" ;;
    refactor*) REACTION="refactor" ;;
    chore*) REACTION="chore" ;;
    docs*) REACTION="docs" ;;
    style*) REACTION="style" ;;
    test*) REACTION="test" ;;
    *) REACTION="update" ;;
  esac

  # =========================
  # 🔹 EXPORT RESULT
  # =========================
  export TYPE
  export DETAIL
  export REACTION
}

# ==========================================
# 🔒 Appreciation Lock
# ------------------------------------------
# Tujuan:
# - Cek apakah hari ini sudah ada commit
# - Gunakan label "commit-today" sebagai lock
# - Mencegah duplicate message (1 hari = 1 event)
#
# Flow:
# 1. Ambil label dari issue tracker
# 2. Cek apakah label "commit-today" ada
# 3. Jika ada → allowed=false (lock aktif)
# 4. Jika tidak → allowed=true (boleh lanjut)
# ==========================================

name: Appreciation Lock
  
permissions:
  issues: read
  
on:
  workflow_call:
    outputs:
      allowed:
        value: ${{ jobs.lock.outputs.allowed }}
        
jobs:
  lock:
    runs-on: ubuntu-latest
    
    outputs:
      allowed: ${{ steps.check.outputs.allowed }}
      
    steps:
      - name: 🔍 Check commit-today label
        id: check
        run: |
          echo "=============================="
          echo "🔒 Appreciation Lock Check"
          echo "=============================="

          ISSUE_NUMBER=1
          LABEL="commit-today"
          REPO="fridfn/workflow-development"

          echo "📦 Repo   : $REPO"
          echo "📌 Issue  : #$ISSUE_NUMBER"
          echo "🏷️ Label  : $LABEL"

          # =========================
          # 📡 FETCH LABELS
          # =========================
          labels=$(curl -s \
            -H "Authorization: token ${{ secrets.ACTIVITY_GITHUB_TOKEN }}" \
            https://api.github.com/repos/$REPO/issues/$ISSUE_NUMBER/labels)

          echo "📥 Labels fetched"

          # =========================
          # 🧠 CHECK EXISTENCE
          # =========================
          already=$(echo "$labels" | jq -r --arg LABEL "$LABEL" '
            [.[] | select(.name == $LABEL)] | length')

          echo "🔎 Found count: $already"

          # =========================
          # ✅ SET RESULT
          # =========================
          if [ "$already" -gt 0 ]; then
            echo "allowed=false" >> $GITHUB_OUTPUT
            echo "⛔ Lock active → skip process"
          else
            echo "allowed=true" >> $GITHUB_OUTPUT
            echo "✅ No lock → process allowed"
          fi

          echo "=============================="
          echo "⏹️ Check Finished"
          echo "=============================="

# ------------------------------------------
# Tujuan:
# - Monitor activity commit setiap jam
# - Kirim reminder jika belum commit
# - Kirim appreciation jika sudah commit
# - Gunakan lock system untuk mencegah duplicate
#
# Flow:
# 1. Check lock (apakah sudah ada commit hari ini)
# 2. Ensure label tersedia
# 3. Check activity hari ini

# 5. Generate message (tone + random)
# 6. Kirim ke Telegram
# 7. Set lock jika sudah commit

name: Daily Commit Reminder (All Repo)

concurrency:
  group: daily-appreciation
  cancel-in-progress: true
  
permissions:
  issues: write
  contents: read
  
on:
  schedule:
    - cron: '0 * * * *' # tiap jam
  workflow_dispatch:
  
jobs:
  lock-check:
    uses: ./.github/workflows/streak.lock.yml
    secrets: inherit
    
  check-activity:
    runs-on: ubuntu-latest
    needs: lock-check
    
    steps:
      - name: 📥 Checkout repository
        uses: actions/checkout@v4

      # 🏷️ ENSURE LABEL
      # =========================

          echo "=============================="
          echo "🏷️ Ensure Label"
          echo "=============================="

          REPO="fridfn/workflow-development"
          LABEL_NAME="commit-today"

          status=$(curl -s -o /dev/null -w "%{http_code}" \
            -H "Authorization: token $TOKEN" \
            https://api.github.com/repos/$REPO/labels/$LABEL_NAME)

          echo "🔍 Label status: $status"

          if [ "$status" -ne 200 ]; then
            echo "➕ Creating label..."

            curl -s -X POST \
              -H "Authorization: token $TOKEN" \
              -H "Accept: application/vnd.github+json" \
              https://api.github.com/repos/$REPO/labels \
              -d '{
                "name": "commit-today",
                "color": "6b00c6",
                "description": "Prevent duplicate appreciation message"
              }'

            echo "✅ Label created"
          else
            echo "✔️ Label already exists"
          fi

      # =========================
      # 🔍 CHECK ACTIVITY
      # =========================
      - name: Check GitHub activity
        id: activity
        run: |

          # =========================

            -H "Authorization: token ${{ secrets.ACTIVITY_GITHUB_TOKEN }}" \
            https://api.github.com/repos/$REPO/issues/$ISSUE_NUMBER/labels/commit-today)

          # =========================
          # 📊 RESULT LOG
          # =========================
          echo "🔍 HTTP Status: $response"

          if [ "$response" = "200" ] || [ "$response" = "204" ]; then
            echo "✅ Label successfully removed"
          elif [ "$response" = "404" ]; then
            echo "⚠️ Label not found (already clean)"
          else
            echo "❌ Failed to remove label"
            echo "🚨 Unexpected response code: $response"
            exit 1
          fi

          echo "⏹️ End     : $(date)"
          echo "=============================="

# ==========================================
# 🌅 Daily Cycle Start
# ------------------------------------------
# Tujuan:
# - Mulai siklus harian
# - Cek apakah kemarin ada commit
# - Kirim pesan pembuka hari ke Telegram
#
# Flow:
# 1. Checkout repo
# 2. Ambil aktivitas GitHub (yesterday)
# 3. Tentukan status commit (true / false)
# 4. Generate message (based on status)
# 5. Kirim ke Telegram
# ==========================================

name: Daily Cycle Start

on:
  schedule:
    - cron: '0 0 * * *' # 00:00 UTC
  workflow_dispatch:
  
jobs:
  start-day:
    runs-on: ubuntu-latest
    
    steps:
      - name: 📥 Checkout repo
        uses: actions/checkout@v4
        
      - name: 🔍 Check Yesterday Activity
        run: |
          echo "=============================="
          echo "🌙 Checking Yesterday Activity"
          echo "=============================="

          username="fridfn"
          yesterday=$(date -u -d "yesterday" +%Y-%m-%d)

          echo "👤 User       : $username"
          echo "📅 Yesterday  : $yesterday"

          # =========================
          # 📡 Fetch GitHub Events
          # =========================
          response=$(curl -s \
            -H "Authorization: token ${{ secrets.ACTIVITY_GITHUB_TOKEN }}" \
            https://api.github.com/users/$username/events)

          # =========================
          # 🧠 Count PushEvent
          # =========================
          yesterday_commit=$(echo "$response" | jq -r --arg y "$yesterday" '
            [.[] 
              | select(.type=="PushEvent") 
              | select(.created_at | startswith($y))
            ] | length')

          echo "📊 Commit Count: $yesterday_commit"

          # =========================
          # ✅ Set Status
          # =========================
          if [ "$yesterday_commit" -gt 0 ]; then
            echo "yesterday_commit=true" >> $GITHUB_ENV
            echo "💜 Status: Active (ada commit)"
          else
            echo "yesterday_commit=false" >> $GITHUB_ENV
            echo "🌙 Status: Idle (tidak ada commit)"
          fi

      - name: 🎲 Generate Morning Message
        run: |
          echo "=============================="
          echo "💬 Generate Morning Message"
          echo "=============================="

          data=$(cat .github/messages/daily.start.json)

          echo "📦 Load message data..."

          # =========================
          # 🎯 Pick Message
          # =========================
          if [ "$yesterday_commit" = "false" ]; then
            echo "⚠️ No commit yesterday → pilih no_commit_yesterday"
            message=$(echo "$data" | jq -r '.no_commit_yesterday[]' | shuf -n 1)
          else
            echo "✅ Commit detected → pilih committed_yesterday"
            message=$(echo "$data" | jq -r '.committed_yesterday[]' | shuf -n 1)
          fi

          echo "📝 Selected Message:"
          echo "$message"

          echo "message=$message" >> $GITHUB_ENV
          
      - name: 📤 Send New Day Message
        env:
          TELEGRAM_TOKEN: ${{ secrets.TELEGRAM_TOKEN }}
          TELEGRAM_CHANNEL_ID: ${{ secrets.TELEGRAM_CHANNEL_ID }}
          
        run: |
          echo "=============================="
          echo "🚀 Sending Message to Telegram"
          echo "=============================="

          curl -s -X POST https://api.telegram.org/bot$TELEGRAM_TOKEN/sendMessage \
            -d chat_id="$TELEGRAM_CHANNEL_ID" \
            --data-urlencode "text=$message" \
            -d "disable_web_page_preview=true"

          echo "✅ Message sent"
==> .github/workflows/git.commit.message.yml <==
# ==========================================
# 📢 Notify Workflow Dev Push
# ------------------------------------------
# Tujuan:
# - Kirim notifikasi ke Telegram saat ada push
# - Bisa dipanggil ulang (reusable workflow)
#
# Trigger:
# - push ke branch main
# - manual (workflow_dispatch)
# - reusable (workflow_call)
#
# Flow:
# 1. Terima input message (private / channel)
# 2. Cek apakah message tersedia
# 3. Kirim ke Telegram (private / channel)
# ==========================================

name: Notify Workflow Dev Push

on:
  workflow_dispatch:

  push:
    branches: [main]
    paths-ignore:
      - README.md
      - .github/*.md

  workflow_call:
    inputs:
      private_message:
        required: false
        type: string
      channel_message:
        required: false
        type: string

    secrets:
      TELEGRAM_TOKEN:
        required: true
      TELEGRAM_CHAT_ID:
        required: false
      TELEGRAM_CHANNEL_ID:
        required: false

jobs:
  notify:
    runs-on: ubuntu-latest

    steps:
      - name: 📤 Send Telegram Message
        env:
          TELEGRAM_TOKEN_OWNER: ${{ secrets.TELEGRAM_TOKEN_OWNER }}
          TELEGRAM_TOKEN_AURIELLE: ${{ secrets.TELEGRAM_TOKEN_AURIELLE }}
          TELEGRAM_CHANNEL_ID: ${{ secrets.TELEGRAM_CHANNEL_ID }}
          TELEGRAM_CHAT_ID: ${{ secrets.TELEGRAM_CHAT_ID }}
          PRIVATE_MESSAGE: ${{ inputs.private_message }}
          CHANNEL_MESSAGE: ${{ inputs.channel_message }}

        run: |
          echo "=============================="
          echo "📢 Telegram Notification Start"
          echo "=============================="

          echo "📦 Repo   : $GITHUB_REPOSITORY"
          echo "🌿 Branch : $GITHUB_REF_NAME"
          echo "👤 Actor  : $GITHUB_ACTOR"

          # =========================
          # 💜 PRIVATE MESSAGE
          # =========================
          if [ ! -z "$PRIVATE_MESSAGE" ]; then
            echo "💬 Sending private message..."

            curl -s -X POST https://api.telegram.org/bot$TELEGRAM_TOKEN/sendMessage \
              -d chat_id=$TELEGRAM_CHAT_ID \
              -d "disable_web_page_preview=true" \
              --data-urlencode "text=$PRIVATE_MESSAGE"

            echo "✅ Private message sent"
          else
            echo "⚠️ No private message"
          fi

          # =========================
          # 📢 CHANNEL MESSAGE
          # =========================
          if [ ! -z "$CHANNEL_MESSAGE" ]; then
            echo "📣 Sending channel message..."

            curl -s -X POST https://api.telegram.org/bot$TELEGRAM_TOKEN/sendMessage \
              -d chat_id=$TELEGRAM_CHANNEL_ID \
              -d "parse_mode=Markdown" \
              -d "disable_web_page_preview=true" \
              --data-urlencode "text=$CHANNEL_MESSAGE"

            echo "✅ Channel message sent"
          else
            echo "⚠️ No channel message"
          fi

          echo "=============================="
          echo "⏹️ Notification Finished"
          echo "=============================="
